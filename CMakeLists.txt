cmake_minimum_required(VERSION 3.16)

# Point d'entrée ESP-IDF — doit être inclus avant project()
include($ENV{IDF_PATH}/tools/cmake/project.cmake)

# Version depuis version.h (source canonique)
file(READ "${CMAKE_CURRENT_LIST_DIR}/main/version.h" _VER_H)
string(REGEX MATCH "define IRRI_VERSION[ \t]+\"([^\"]+)\"" _ ${_VER_H})
set(PROJECT_VER ${CMAKE_MATCH_1})

# Lire le mode PROD depuis sdkconfig avant project() (CONFIG_* pas encore chargé)
set(_IRRI_PROD OFF)
set(_MACHINE_ID "")
set(_CANON_ID "")
if(EXISTS "${CMAKE_CURRENT_LIST_DIR}/sdkconfig")
    file(READ "${CMAKE_CURRENT_LIST_DIR}/sdkconfig" _SDKCONFIG)
    if(_SDKCONFIG MATCHES "CONFIG_IRRI_PROD=y")
        set(_IRRI_PROD ON)
        if(_SDKCONFIG MATCHES "CONFIG_IRRI_MACHINE_ST1BIS_82_330=y")
            set(_MACHINE_ID "st1bis_82_330")
        endif()
        if(_SDKCONFIG MATCHES "CONFIG_IRRI_CANON_SR100C=y")
            set(_CANON_ID "sr100c")
        elseif(_SDKCONFIG MATCHES "CONFIG_IRRI_CANON_SR150C=y")
            set(_CANON_ID "sr150c")
        endif()
    endif()
endif()

if(_IRRI_PROD AND _MACHINE_ID AND _CANON_ID)
    set(FIRMWARE_NAME "irrifrance_${_MACHINE_ID}_${_CANON_ID}_v${PROJECT_VER}")
    message(STATUS "Build PROD : ${FIRMWARE_NAME}.bin")
    message(STATUS "  Version  : ${PROJECT_VER}")
else()
    set(FIRMWARE_NAME "irrifrance_dev")
    message(STATUS "Build DEV  : ${FIRMWARE_NAME}.bin  v${PROJECT_VER}")
endif()

project(${FIRMWARE_NAME})
